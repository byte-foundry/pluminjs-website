var demo = new p.Font({ familyName: 'Demo' }),
    glyph_p = new p.Glyph({
        name: 'p',
        advanceWidth: 512
    });

glyph_p.addContour(new p.Path.Rectangle({
    point: [0, -255],
    size: [100, 800]
}));
glyph_p.addContour(new p.Path.Ellipse({
    point: [50, 0],
    size: [400, 550]
}));
var counter = new p.Path.Ellipse({
    point: [100, 100],
    size: [250, 350]
});
counter.reverse();
glyph_p.addContour(counter);

demo.addGlyph(glyph_p)
    .prepareOT()
    .addToFonts();
